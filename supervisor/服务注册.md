# 服务注册
```shell
# 到对应配置目录
cd /data/models/supervisor
# 
sudo cp ls.service    /usr/lib/systemd/system/ls.service
sudo systemctl enable ls
sudo systemctl status ls
# 重启
sudo systemctl restart ls
#  停止
systemctl stop ls

# 服务配置更新
sudo systemctl daemon-reload

# 重启AI
sudo /home/beyond/.conda/envs/yolo11/bin/supervisorctl -c /data/models/supervisor/supervisord.conf restart ai
```
